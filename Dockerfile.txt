# Etapa 1: build de la app (Vite/React/TS)
FROM node:20-alpine AS build

WORKDIR /app

# Copiamos solo lo necesario para instalar deps
COPY package*.json ./
RUN npm install

# Copiamos el resto del proyecto y construimos
COPY . .
RUN npm run build

# Etapa 2: servidor web Nginx para archivos estáticos
FROM nginx:alpine

# Copiamos el build al directorio público de nginx
COPY --from=build /app/dist /usr/share/nginx/html

# Puerto expuesto dentro del contenedor
EXPOSE 80

CMD ["nginx", "-g", "daemon off;"]
